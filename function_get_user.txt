CREATE FUNCTION [PMS].[getUserV2] (@userName nvarchar(255), @password integer)
returns integer
AS
BEGIN
DECLARE @resultExistence integer
DECLARE @resultLogin integer

	SET @resultExistence = (SELECT  USUARIOS.Id_Usuario FROM Pms.USUARIOS where User_Nombre like @userName)

	if @resultExistence >= 0 
	BEGIN
	SET @resultLogin = ( SELECT USUARIOS.Id_Usuario FROM Pms.USUARIOS where User_Password = CONVERT(BINARY(1),@password) AND User_Nombre like @userName);
	
	END
	else if @resultExistence IS NULL
	SET @resultLogin = -2;

	return @resultLogin

END
